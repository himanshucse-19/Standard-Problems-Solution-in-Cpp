#include <bits/stdc++.h>
using namespace std;
int main()
{
    int n;
    cin>>n;
    int a[n];
    for(int i=0;i<n;i++)
    {
      cin>>a[i];
    }
    for(int i=0;i<n;i++)
    {
      if(a[i]!=(i+1))
      {
        swap(a[i],a[a[i]-1]);
      }
    }
    for(int i=0;i<n;i++)
    {
      if(a[i]!=(i+1))
      {
        cout<<a[i]<<" "<<(i+1)<<endl;
      }
    }
    //where a[i] is duplicate and (i+1) is missing
    return 0;
}
