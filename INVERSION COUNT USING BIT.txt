lli BIT[200005];
void update(lli i,lli val,lli n)
{
    while(i<=n)
    {
        BIT[i]+=val;
        i+=((i&(-i)));

}
lli query(lli i)
{
    lli ans=0;
    while(i>0)
    {
        ans+=BIT[i];
        i=(i&(i-1));
    }
    return ans;
}
